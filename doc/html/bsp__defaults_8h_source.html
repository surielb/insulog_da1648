202 /*
203  * Maximum sleep time the sleep timer supports
204  *
205  * dg_configTim1Prescaler can be zero. if dg_configTim1Prescaler is not zero then
206  * ( dg_configTim1Prescaler + 1 ) MUST be a power of 2! */
207 #if (dg_configUSE_LP_CLK == LP_CLK_32000)
208 #define dg_configTim1Prescaler (3)
209 #define dg_configMaxSleepTime (8) // sec
210 #elif (dg_configUSE_LP_CLK == LP_CLK_32768)
211 #define dg_configTim1Prescaler (3)
212 #define dg_configMaxSleepTime (8) // sec
213 #elif (dg_configUSE_LP_CLK == LP_CLK_RCX)
214 #define dg_configTim1Prescaler (0)
215 #define dg_configMaxSleepTime (8) // sec
216 #elif (dg_configUSE_LP_CLK == LP_CLK_ANY)
217 /* Assuming that the frequency of the external digital clock is 32000Hz,
218  * period: 1/32000 = 31.25usec
219  * Timer1 wrap-around time: 65536 * 31.25 = 2.048sec
220  * With a prescaler equal to 3, the wrap around time becomes: (1 + 3) * 2.048 = ~8sec.
221  * If the clock frequency is too slow, i.e. 16000Hz then
222  * period: 1/16000 = 62.5usec
223  * Timer1 wrap-around time = 65536 * 62.5 = 4.096sec
224  * and the prescaler should be 1 so that: (1 + 1) * 4.096 = ~8sec.
225  * If the clock frequency is too high, i.e. 125000Hz then
226  * period: 1/125000 = 8usec
227  * Timer1 wrap-around time = 65536 * 8 = 0.524sec
228  * and the prescaler should be 3 so that: (1 + 3) * 0.524 = ~2sec.
229  * The values MUST be defined in the custom_config_<>.h file!
230  */
231 #else
232 #error "dg_configUSE_LP_CLK has invalid setting"
233 #endif
